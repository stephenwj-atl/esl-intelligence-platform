MODULE 2 — CURRENT STAGE ENGINE
Create function:
function determineStage(project) {
  if (!project.screened) return 1
  if (!project.baselineComplete) return 2
  if (!project.riskCalculated) return 3
  if (!project.decisionMade) return 4
  if (!project.conditionsDefined) return 5
  if (!project.validationComplete) return 6
  return 7
}
